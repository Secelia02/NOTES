# Move line down
- Alt + down arrow

# Move line up
- Alt + up arrow

# Copy line up
- Shift + Alt + up arrow

# Copy line down
- Shift + Alt + down arrow

# Insert line below
- Ctrl + Enter

# Jump to matching brackets
- CTRL + Shift + \

# Indent line
- CTRL + ]

# Outdent line
- CTRL + [

# Go to the beginning of file
- CTRL + Home

# Go to the end of file
- CTRL + End

# Toggle line comment
- CTRL + /

# Find
- CTRL + F

# Replace
- CTRL + H

# Find next
- F3

# Find previous
- Shift + F3

# Go to line
- CTRL + G

# Go to file
- CTRL + P
